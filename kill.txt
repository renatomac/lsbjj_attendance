# Find all Python processes
ps aux | grep python

# Look for processes with "manage.py" or "runserver"
# Then kill them specifically:

# Kill all runserver processes
pkill -f "manage.py runserver"

# If that doesn't work, try a more aggressive approach
pkill -9 -f "python manage.py"

# Also kill any other Python processes that might be holding the port
pkill -9 -f "gunicorn"  # if you're using gunicorn

pkill -f "manage.py runserver" || true


# Find the process using port 8000
sudo lsof -i :8000

# Or use netstat
sudo netstat -tulpn | grep :8000

# Kill the process (replace PID with the actual process ID)
sudo kill -9 <PID>
